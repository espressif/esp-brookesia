set(COMPONENT_DIR ${CMAKE_CURRENT_SOURCE_DIR})
set(COMPONENT_SRC_DIR ${COMPONENT_DIR}/src)
set(COMPONENT_INCLUDE_DIRS ${COMPONENT_DIR}/include)
set(COMPONENT_PRIVATE_INCLUDE_DIRS ${COMPONENT_SRC_DIR})
set(COMPONENT_REQUIRES brookesia_gui_interface brookesia_lib_utils brookesia_hal_interface brookesia_service_display brookesia_service_manager)
set(COMPONENT_SRCS_C "")
set(COMPONENT_SRCS_CPP "")
set(COMPONENT_SRCS_C_COMPILE_FLAGS "")
set(COMPONENT_SRCS_CPP_COMPILE_FLAGS "")

file(GLOB_RECURSE GUI_LVGL_SRCS_C ${COMPONENT_SRC_DIR}/*.c)
file(GLOB_RECURSE GUI_LVGL_SRCS_CPP ${COMPONENT_SRC_DIR}/*.cpp)
file(GLOB_RECURSE GUI_LVGL_PORT_SRCS_CPP ${COMPONENT_SRC_DIR}/port/*/*.cpp)
list(REMOVE_ITEM GUI_LVGL_SRCS_CPP ${GUI_LVGL_PORT_SRCS_CPP})
if(ESP_PLATFORM)
    file(GLOB_RECURSE GUI_LVGL_PLATFORM_SRCS_CPP ${COMPONENT_SRC_DIR}/port/esp/*.cpp)
elseif(EMSCRIPTEN)
    file(GLOB_RECURSE GUI_LVGL_PLATFORM_SRCS_CPP ${COMPONENT_SRC_DIR}/port/wasm/*.cpp)
else()
    file(GLOB_RECURSE GUI_LVGL_PLATFORM_SRCS_CPP ${COMPONENT_SRC_DIR}/port/sdl/*.cpp)
endif()
list(APPEND COMPONENT_SRCS_C ${GUI_LVGL_SRCS_C})
list(APPEND COMPONENT_SRCS_CPP ${GUI_LVGL_SRCS_CPP})
list(APPEND COMPONENT_SRCS_CPP ${GUI_LVGL_PLATFORM_SRCS_CPP})

include(${CMAKE_CURRENT_LIST_DIR}/cmake/image_pack.cmake)

if(ESP_PLATFORM)
    include(${CMAKE_CURRENT_LIST_DIR}/cmake/esp_platform.cmake)
else()
    include(${CMAKE_CURRENT_LIST_DIR}/cmake/pc_platform.cmake)
endif()

    set_source_files_properties(${COMPONENT_SRCS_C}
        PROPERTIES
            COMPILE_FLAGS "${COMPONENT_SRCS_C_COMPILE_FLAGS}"

    set_source_files_properties(${COMPONENT_SRCS_CPP}
        PROPERTIES
            COMPILE_FLAGS "${COMPONENT_SRCS_CPP_COMPILE_FLAGS}"
    )
endif()
